238 	protected <S> List<S> createList(List<Object> solutions) {
239 		List<S> list = new ArrayList<S>();
240 		for (Object solution : solutions) {
241 			if (solution instanceof Object[]) {
242 				Object[] objects = (Object[]) solution;
243 				if (objects.length > 0) {
244 					Object object = objects[0];
245 					S s = (S) object;
246 					list.add(s);
247 				}
248 			}
249 		}
250 		return list;
251 	}
